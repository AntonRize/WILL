\begin{align}
    \kappa_\oplus^2 &= \frac{2 R_\oplus g}{c^2} \\
    R_{s\oplus} &= \kappa_\oplus^2 R_\oplus = \frac{2 R_\oplus^2 g}{c^2} \approx 0.00887 \text{ m}
\end{align}

\noindent \textbf{2. Earth-Moon Distance ($a_M$)} \\
Inputs: Moon sidereal period $T_M = 2.3606 \times 10^6$ s.

    a_M = \left( \frac{R_{s\oplus} c^2}{8\pi^2} \right)^{1/3} T_M^{2/3} \approx 3.83 \times 10^8 \text{ m}
\end{equation}

Inputs: Lunar dichotomy angle $\Phi \approx 89.853^\circ$.
\begin{equation}
    a_\oplus = \frac{a_M}{\cos \Phi} \approx 1.496 \times 10^{11} \text{ m}
\end{equation}

Inputs: Earth sidereal period $T_\oplus = 31557600$ s.
\begin{equation}
    R_{s\odot} = \frac{8\pi^2 a_\oplus^3}{T_\oplus^2 c^2} \approx 2953 \text{ m}
\end{equation}

\noindent \textbf{5. Mercury's Relational Projections} \\

\begin{equation}
    \Delta\varphi_{\text{century}} = \Delta\varphi \times \frac{36525 \text{ days}}{87.969 \text{ days}} \times \frac{180 \times 3600}{\pi} \approx 43.01 \text{ arcsec/cy}
