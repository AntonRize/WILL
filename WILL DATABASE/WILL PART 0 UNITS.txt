\[
E = \hbar\omega, \quad [E] = [m/s].

\subsubsection{Quantum Harmonic Oscillator}
\[
\hat{H} = -\frac{\hbar^2}{2m}\nabla^2 + \frac{1}{2}kx^2,
\]
where:
\begin{itemize}
    \item $k$: stiffness, $[k] = \left[\frac{1}{ms}\right]$,
    \item $x$: position, $[x] = [m]$.
\end{itemize}
