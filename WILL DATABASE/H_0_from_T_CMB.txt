\section{Step-by-Step Derivation}

\subsection{Step 1: Radiation Density Calculation ($\rho_{\gamma}$)}
\textit{Meaning:} Determining the absolute mass-energy density of the photon gas filling the universe. This provides the "Energy" input for the $Spacetime \equiv Energy$ equivalence.

\begin{equation}
    \rho_{\gamma} = \frac{4 \sigma_{SB} T_0^4}{c^3}
\end{equation}
Using the input $T_0 = 2.7255$ K:
\begin{equation}
    \rho_{\gamma} \approx 4.641 \times 10^{-31} \, \text{kg/m}^3
\end{equation}

\subsection{Step 2: Maximal Geometric Density ($\rho_{max}$)}
\textit{Meaning:} Calculating the saturation density of the geometric field.
\textit{Hypothesis:} The relationship between the radiation background ($\rho_{\gamma}$) and the structural limit ($\rho_{max}$) is mediated by the geometry of the ground state ($\alpha$). Since energy scales as the square of the projection ($E \sim \beta^2$), the scaling factor is $\alpha^2$.

\begin{equation}
    \rho_{max} = \frac{\rho_{\gamma}}{\alpha^2}
\end{equation}
Using $\alpha \approx 7.297 \times 10^{-3}$ ($\alpha^2 \approx 5.325 \times 10^{-5}$):
\begin{equation}
    \rho_{max} \approx \frac{4.641 \times 10^{-31}}{5.325 \times 10^{-5}} \approx 8.715 \times 10^{-27} \, \text{kg/m}^3
\end{equation}

\subsection{Step 3: Vacuum Energy Density ($\rho_{\Lambda}$)}
\textit{Meaning:} Determining the effective density of the vacuum that drives expansion.
\textit{Logic (WILL RG Part B I):} In a closed WILL system, the potential sector ($\kappa^2$) accounts for $2/3$ of the total resource ($\Omega_{pot} = 2/3$).

\begin{equation}
    \rho_{\Lambda} = \Omega_{pot} \cdot \rho_{max} = \frac{2}{3} \rho_{max}
\end{equation}
\begin{equation}
    \rho_{\Lambda} \approx \frac{2}{3} (8.715 \times 10^{-27}) \approx 5.810 \times 10^{-27} \, \text{kg/m}^3
\end{equation}

\subsection{Step 4: The Hubble Constant ($H_0$)}
\textit{Meaning:} Converting the vacuum tension into the expansion rate parameter.
\textit{Logic:} Using the standard Friedman relation for a flat universe dominated by this vacuum density. The geometric relation simplifies to:

\begin{equation}
    H_0 = \sqrt{4\pi G \rho_{\Lambda}}
\end{equation}
